# Status: pending

# Description: Add support for sending messages to multiple FCM tokens simultaneously.
# Details:
1. Update node properties to allow input of multiple tokens
2. Modify the execute method to handle multiple tokens
3. Use the Firebase Admin SDK's sendMulticast method for efficient multi-device messaging

Example implementation:
```typescript
// ... (inside execute method)
const tokens = this.getNodeParameter('tokens', 0) as string[];

const message: messaging.MulticastMessage = {
  tokens,
  notification,
  android,
  data,
};

try {
  const response = await messaging().sendMulticast(message);
  returnData.push({
    json: {
      success: true,
      successCount: response.successCount,
      failureCount: response.failureCount,
      responses: response.responses,
    },
  });
} catch (error) {
  throw new Error(`Failed to send multicast message: ${error.message}`);
}
```
