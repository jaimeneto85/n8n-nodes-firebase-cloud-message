Example implementation:
```typescript
import { getMessaging } from 'firebase-admin/messaging';

// ... (inside execute method)
const useInAppMessaging = this.getNodeParameter('useInAppMessaging', 0) as boolean;

if (useInAppMessaging) {
  const campaignName = this.getNodeParameter('inAppCampaignName', 0) as string;
  const analyticsLabel = this.getNodeParameter('analyticsLabel', 0) as string;

  try {
    await getMessaging().sendToDevice(token, {
      data: {
        'fiam-campaign-name': campaignName,
        'fiam-analytics-label': analyticsLabel,
      },
    });
    logger.info('In-App Messaging campaign triggered', { campaignName });
  } catch (error) {
    throw new FCMError(`Failed to trigger In-App Messaging: ${error.message}`);
  }
}
```
